## Internal Caller

```ts twoslash
import { os, createRouterCaller } from '@orpc/server'

const router = os.router({
    ping: os.func(() => 'pong')
})

const caller = createRouterCaller({
    router: router,
    context: {},
})

const result = await caller.ping(undefined) // result is 'pong'
```
